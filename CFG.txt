<program> ::= <instruction_list>

<instruction_list> ::= <instruction> | <instruction> <instruction_list>

<instruction> ::= <comment>
              | <load_address>
              | <move>
              | <jump_link>
              | <load_immediate>
              | <syscall>
              | <jump>

<comment> ::= 'COMMENT' <NEWLINE>

<load_address> ::= 'LOAD_ADDRESS' <REGISTER> ',' <NUMBER> <NEWLINE>

<move> ::= 'MOVE' <REGISTER> ',' <REGISTER> <NEWLINE>

<jump_link> ::= 'JUMPLINK' <REGISTER> ',' <REGISTER> <NEWLINE>

<load_immediate> ::= 'LOAD_IMMEDIATE' <REGISTER> ',' <NUMBER> <NEWLINE>

<syscall> ::= 'SYSCALL' <NEWLINE>

<jump> ::= 'JUMP' <REGISTER> <NEWLINE>

<REGISTER> ::= 'REGISTER' | 'LPAREN' 'REGISTER' 'RPAREN'

<NUMBER> ::= 'NUMBER'

<NEWLINE> ::= 'NEWLINE'
